const checkNameAndTag = ({
  match,
  params,
}: {
  match: any;
  params: PlayerParams;
}) => {
  match.info.participants.forEach((item) => {
    if (
      item.riotIdGameName === params.username &&
      item.riotIdTagline === params.tagLine
    ) {
      return true;
    }
  });
};
